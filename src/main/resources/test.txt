lettype subfunc = vec2d -> vec3d;

let switch2d: subfunc = /\ v2:vec2d. {
    {x1=v2.x2, x2=v2.x1};
} -> vec3d;


// {x1:Int, x2:Int} -> {x1:Int, x2:Int, x3:Int}

    let v2:vec2d = {x1=1, x2=2} in f (v2);
